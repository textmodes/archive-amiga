proc nosoul {} {
  set file [open nosoul.txt r]
  while {![eof $file]} {
    set line [gets $file]
    if {[string length $line] > 0} {
      append nosoul $line
    }
  } ; close $file
  foreach soulLINE nosoul {
    puts stdout "$soulLINE"
  }
}
